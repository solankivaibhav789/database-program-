CREATE OR REPLACE
PACKAGE BODY PK1
AS
BOOLEAN FLAG;
VDEPTNO EMP.DEPTNO%TYPE;
C NUMBER(3);
FUNCTION F2(E IN EMP.EMPNO%TYPE,D IN EMP.DEPTNO%TYPE) RETURN BOOLEAN
AS
BEGIN
	SELECT DEPTNO INTO VDEPTNO
	FROM EMPNO
	WHERE EMPNO = E;
	
	IF(VDEPNO = D)THEN
		RETURN TRUE;
	ELSE
		RETURN FALSE;
	END IF;
END F2;
PROCEDURE P2(E IN EMP.EMPNO%TYPE,D IN EMP.DEPTNO%TYPE)
AS
BEGIN
	FLAG = F2(E,D);
	IF(FLAG = TRUE)THEN
		SELECT COUNT(*) INTO C
		FROM EMP
		WHERE DEPNO = D;
	ELSE
		C = NULL;
	END IF;
	DBMS_OUTPUT.PUT_LINE('TOTAL EMPLOYEE ON DEPTNO ' || DEPTNO || ' IS ' || C);
END;
